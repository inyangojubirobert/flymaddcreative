🆕 Migration: participant_profiles

Stores each participant's public bio, media (video / photo / link), and optional contact info.

-- ─────────────────────────────────────────────────────────────────
-- participant_profiles
-- One row per participant (UNIQUE on participant_id).
-- POST /api/onedream/profile/[username] does an upsert on this table.
-- ─────────────────────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS participant_profiles (
    id               UUID         DEFAULT gen_random_uuid() PRIMARY KEY,
    participant_id   UUID         NOT NULL UNIQUE REFERENCES participants(id) ON DELETE CASCADE,

    -- Bio / description
    bio              TEXT         DEFAULT '',

    -- Media  (video upload, photo upload, YouTube embed, or any external link)
    media_type       TEXT         CHECK (media_type IN ('video_upload','photo_upload','youtube','external')),
    media_url        TEXT,           -- public URL (Supabase Storage or external)
    storage_path     TEXT,           -- Supabase Storage object path (uploads only)
    media_title      TEXT         DEFAULT '',

    -- Contact info (always stored; shown publicly only when contact_is_public = TRUE)
    phone            TEXT         DEFAULT '',
    contact_email    TEXT         DEFAULT '',
    whatsapp         TEXT         DEFAULT '',
    instagram        TEXT         DEFAULT '',
    facebook         TEXT         DEFAULT '',
    twitter          TEXT         DEFAULT '',
    website          TEXT         DEFAULT '',
    contact_is_public BOOLEAN     DEFAULT FALSE,

    created_at       TIMESTAMPTZ  DEFAULT NOW(),
    updated_at       TIMESTAMPTZ  DEFAULT NOW()
);

-- Fast lookup by participant
CREATE INDEX IF NOT EXISTS idx_participant_profiles_participant_id
    ON participant_profiles (participant_id);

-- Auto-bump updated_at on every update
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN NEW.updated_at = NOW(); RETURN NEW; END;
$$;

CREATE OR REPLACE TRIGGER trg_participant_profiles_updated_at
    BEFORE UPDATE ON participant_profiles
    FOR EACH ROW EXECUTE FUNCTION set_updated_at();

-- Row-level security
ALTER TABLE participant_profiles ENABLE ROW LEVEL SECURITY;

-- Anyone can read profiles (public page)
CREATE POLICY "Public read participant_profiles"
    ON participant_profiles FOR SELECT USING (TRUE);

-- Backend API routes use the SERVICE ROLE key and bypass RLS for writes

🪣 Supabase Storage Buckets

Create two public buckets in Supabase → Storage:
  • participant-videos — for uploaded video files (max 100 MB per file)
  • participant-photos — for uploaded image files (max 10 MB per file)
Set both to Public so generated URLs are accessible without auth. 
-- Or via SQL (run in Supabase SQL Editor):
INSERT INTO storage.buckets (id, name, public)
VALUES
    ('participant-videos', 'participant-videos', true),
    ('participant-photos', 'participant-photos', true)
ON CONFLICT (id) DO NOTHING;
